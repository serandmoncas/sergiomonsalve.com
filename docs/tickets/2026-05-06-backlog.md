**sitemap.ts:**
```ts
import { MetadataRoute } from 'next'

export default function sitemap(): MetadataRoute.Sitemap {
  const base = 'https://sergiomonsalve.com'
  const routes = ['', '/about', '/contact']
  const locales = ['es', 'en']
  return locales.flatMap(locale =>
    routes.map(route => ({
      url: `${base}/${locale}${route}`,
      lastModified: new Date(),
      changeFrequency: 'monthly' as const,
      priority: route === '' ? 1 : 0.8,
    }))
  )
}
```

**robots.ts:**
```ts
import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: { userAgent: '*', allow: '/' },
    sitemap: 'https://sergiomonsalve.com/sitemap.xml',
  }
}
```

### TICKET-013 — Comments on blog posts

**Schema:**
```sql
create table comments (
  id           uuid primary key default gen_random_uuid(),
  post_slug    text not null,
  author_name  text not null,
  author_email text not null,
  body         text not null,
  approved     boolean not null default false,
  created_at   timestamptz not null default now()
);
-- Public can insert (unapproved by default)
-- Only admin can read all, approve, delete
```

**Flow:** Submit → saved with `approved=false` → admin reviews at `/admin/comments` → approves → appears on post.

Spam protection: honeypot field `website` (hidden, must be empty) + Zod validation on API route.
